import rclpy
from rclpy.node import Node

class Minimal(Node):
    def __init__(self):
        super().__init__('minimal_node')
        self.get_logger().info('Node inicializado')

def main():
    rclpy.init()
    node = Minimal()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
