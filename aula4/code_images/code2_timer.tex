    def __init__(self):
        super().__init__('timer_example')
        self.i = 0
        self.timer = self.create_timer(0.5, self.on_timer)  # 2 Hz

    def on_timer(self):
        self.i += 1
        self.get_logger().info(f'Tick #{self.i}')
